303  const size_t samplesGot = chunk.get_sample_count();
304  if (remaining < samplesGot) {

311  if (raw != NULL) {

313  PFC_ASSERT( rawSize % samplesGot == 0 );
314  raw->set_size( (t_size) ( (t_uint64) rawSize * remaining / samplesGot ) );
315  }

318  }
